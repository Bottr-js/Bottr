import Bottr;

var Bot = new Bottr.Bot()

bot.on('message_received', (message, session) => {

  session.send('The World's Simplest Bot Framework', {
    buttons: [
      { title: 'Get Started', href: 'https://bottr.co/docs/install' },
      { title: 'Contribute', href: 'https://github.com/Bottr-js/Bottr' }
    ]
  })

  session.send({
    url: 'images/code.png'
  })
})

bot.listen()
